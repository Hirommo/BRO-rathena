-	script	tkreset#smart	-1,{

	bindatcmd("tkreset", strnpcinfo(0) +"::OnTK");
	end;
OnTK:

		message strcharinfo(0), "Mapa resetado. Relogue!.";

	}

	set PC_HATE_MOB_STAR, 0;
	set PC_HATE_MOB_SUN, 0;

	end;
}
